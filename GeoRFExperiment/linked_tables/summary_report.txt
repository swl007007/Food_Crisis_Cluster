5. USAGE INSTRUCTIONS

   links = pd.read_csv('linked_tables/table_links.csv')
   partition_file = links[links['name'] == name]['partition_table_path'].iloc[0]
   partition_df = pd.read_csv(f'linked_tables/{partition_file}')
